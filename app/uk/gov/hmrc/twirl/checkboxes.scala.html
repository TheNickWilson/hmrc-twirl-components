@( field: Field, legend: String, options: Seq[InputOption], hiddenLegend: Boolean = false, hint: Option[String] = None, fieldsetClasses: Set[String] = Set.empty, legendClasses: Set[String] = Set("bold-small") )(implicit messages: Messages) @fieldset( field = field, legend = legend, hiddenLegend = hiddenLegend, hint = hint, fieldsetClasses = fieldsetClasses, legendClasses = legendClasses ) { @for((option, index) <- options.zipWithIndex) { @checkbox( field = field, label = option.label, value = option.value, index = Some(index), dataTarget = option.dataTarget ) } }
